    private void BtnSalvarActionPerformed(java.awt.event.ActionEvent evt) {//GEN-FIRST:event_BtnSalvarActionPerformed
       
        
       MesaController mc = new MesaController();
       
       String s = comboBoxStatus.getSelectedItem().toString();
       String y = String.valueOf(s.charAt(0));
       int status = Integer.parseInt(y);
       
       
       
        try {
           if(txtDescricao.getText().equals("") || status == 2){
               JOptionPane.showMessageDialog(null, "Descricao vazia ou status invalido");
           }
           else{
               mc.salvar(txtDescricao.getText(), status);
               JOptionPane.showMessageDialog(this, "Mesa salva com sucesso!");
               statusLists = new StatusController().listaStatus();
           }
        
        } catch (SQLException e) {
            JOptionPane.showMessageDialog(this, "Não foi possivel salvar mesa!\n" + e.getLocalizedMessage());
        } catch (Exception ex) {
            Logger.getLogger(MesaView.class.getName()).log(Level.SEVERE, null, ex);
        } 

       LimparCampos();
      // }
    }//GEN-LAST:event_BtnSalvarActionPerformed

    private void btnAlterarActionPerformed(java.awt.event.ActionEvent evt) {//GEN-FIRST:event_btnAlterarActionPerformed
        if(!txtDescricao.getText().equals("")){
            MesaController cc = new MesaController();
            int mesa_id = mesa.getIdMesa();
            try {

                String n = comboBoxStatus.getSelectedItem().toString();
                String x = String.valueOf(n.charAt(0));
                int statusID = Integer.parseInt(x);

                cc.alterar(mesa_id, txtDescricao.getText(), statusID);


                JOptionPane.showMessageDialog(null, "Mesa alterada com sucesso!");
              //  LimparCampos();
           
        
            } catch (SQLException e) {
                JOptionPane.showMessageDialog(this, "Não foi possivel alterar a mesa!\n" + e.getLocalizedMessage());
            }
        }else try {
            throw new SQLException("Descricao Invalida");
        } catch (SQLException ex) {
            Logger.getLogger(MesaView.class.getName()).log(Level.SEVERE, null, ex);
        }
    }//GEN-LAST:event_btnAlterarActionPerformed

    private void btnExcluirActionPerformed(java.awt.event.ActionEvent evt) {//GEN-FIRST:event_btnExcluirActionPerformed
        
        MesaController ec = new MesaController();
        try {
            ec.excluir(Integer.parseInt(txtCodigo.getText()));
            JOptionPane.showMessageDialog(this, "Mesa excluída com sucesso!");
            btnAlterar.setEnabled(false);
            this.setVisible(false);
           
            LimparCampos();           
            

        } catch (SQLException e) {
            JOptionPane.showMessageDialog(this, "Não foi possivel excluir o mesa!\n" + e.getLocalizedMessage());
        }
    }//GEN-LAST:event_btnExcluirActionPerformed

  public void recebeTitulo(int mesa_id) throws SQLException{
     
     mesa = new MesaController().findMesaPorId(mesa_id);
     
     
     txtCodigo.setText(String.valueOf(mesa.getIdMesa()));
     txtDescricao.setText(mesa.getDescricao());
    
      
     for( Status status: statusLists ){
          
          if(mesa.getStatus()==status.getIdStatus()){
              comboBoxStatus.setSelectedItem(status.getIdStatus()+" - "+status.getStatusDescricao());
          }
          
      }
             
     BtnSalvar.setEnabled(false);
     btnAlterar.setEnabled(true);
     btnExcluir.setEnabled(true);
    // comboBoxCandidato.set
      
      
      
  }
